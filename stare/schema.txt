Table students {
  id integer [primary key]
  groupId integer
  firstName string
  lastName string
  birthday date
  genderId integer
}

Table genders {
  id integer [primary key]
  value string
}

Table teachers {
  id integer [primary key]
  firstName string
  lastName string
  birthday date
  phoneNumber string
  email string
  additionalInfo string
}

Table parents {
  id integer [primary key]
  firstName string
  lastName string
  phoneNumber string
  email string
}

Table parents_students {
  id integer [primary key]
  parentId integer
  studentId integer
}

Table classrooms {
  id integer [primary key]
  location string
}

Table groups {
  id integer [primary key]
  yearId integer
  home_teacher_id integer
}

Table marks {
  id integer [primary key]
  subjectId integer
  studentId integer
  value integer
  comment string
  weight integer
}

Table years {
  id integer [primary key]
  value integer
}

Table attendances {
  id integer [primary key]
  datexd datetime
  lessonId integer
}

Table attendance_student {
  id integer [primary key]
  attendanceId integer
  studentId integer
  present bool
}

Table subjects {
  id integer [primary key]
  shortName string
  longName string

}

Table lessons {
  id integer [primary key]
  teacherId integer
  subjectId integer
  groupId integer
  hourId integer
  classroomId integer
  dayId integer
}

Table hours {
  id integer [primary key]
  start_hour integer
  start_minutes integer
  end_hour integer
  end_minutes integer
}

Table days {
  id integer [primary key]
  value string
}

Ref: students.genderId > genders.id
Ref: classrooms.id < lessons.classroomId
Ref: students.groupId > groups.id

Ref: parents_students.parentId > parents.id
Ref: parents_students.studentId > students.id

Ref: groups.home_teacher_id > teachers.id
Ref: groups.yearId > years.id
Ref: groups.id < lessons.groupId

Ref: marks.subjectId > subjects.id
Ref: marks.studentId > students.id

Ref: attendance_student.attendanceId > attendances.id
Ref: attendance_student.studentId > students.id

Ref: attendances.lessonId > lessons.id

Ref: lessons.teacherId > teachers.id
Ref: lessons.subjectId > subjects.id
Ref: lessons.hourId > hours.id
Ref: lessons.dayId > days.id

Table contracts {
  id integer [primary key]
  studentId integer
  parentId integer
  startDate date
  endDate date
  monthlyAmount decimal
}

Table payments {
  id integer [primary key]
  contractId integer
  dueDate date
  paidDate date
  amount decimal
  status string 
}

Ref: contracts.studentId > students.id
Ref: contracts.parentId > parents.id
Ref: payments.contractId > contracts.id

// ------------------------------------------

Table remark {
  id integer [primary key]
  studentId integer
  teacherId integer
  value string
}

Ref: remark.studentId > students.id
Ref: remark.teacherId > teachers.id
